#version 460 core

out vec4 FragColor;

in vec2 texCoords;
in vec3 Normals;
in vec3 FragPos;

uniform vec3 LightColor;

void main()
{
       vec3 result = LightColor;

//HDR
       result = vec3(1.0) - exp(-result * 1.0);
//Gamma
       result = pow(result, vec3(1 / 2.2));
       FragColor = vec4(result, 1.0);      
}

#version 460 core

uniform vec3 fragColor = vec3(1.0, 1.0, 1.0);
uniform bool image = false;
uniform sampler2D imageTex;

out vec4 outColor;

in vec2 texCoords;

void main()
{
     if(image)
     {
       outColor = texture(imageTex, texCoords);
     }
     else{
       outColor = vec4(fragColor ,1.0);
     }
}
